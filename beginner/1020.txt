import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        
        int days = sc.nextInt();
        
        int years = days / 365;
        days %= 365;
        int months = days / 30;
        days %= 30;
        
        // Ensure output matches expected Beecrowd format
        System.out.printf("%d year(s)%n", years);
        System.out.printf("%d month(s)%n", months);
        System.out.printf("%d day(s)%n", days);
        
        sc.close();
    }
}
